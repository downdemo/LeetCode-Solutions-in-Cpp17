* 动态规划，除非利润为负，否则必须有一次买入卖出，`dp[i]` 表示第 `i + 1` 天卖出可获得的最大利润，如果最大利润为负，则无买入卖出

```cpp
class Solution {
 public:
  int maxProfit(vector<int>& prices) {
    if (prices.size() <= 1) {
      return 0;
    }
    vector<int> dp(prices.size());
    int res = 0;
    for (int i = 1; i < dp.size(); ++i) {
      dp[i] = max(dp[i - 1] + prices[i] - prices[i - 1], 0);
      res = max(res, dp[i]);
    }
    return res;
  }
};
```
