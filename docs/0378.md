* 每行每列均为升序，第一行第一个元素即为最小元素，最后一行最后一列即为最大元素，在最小元素到最大元素中二分查找即可

```cpp
class Solution {
 public:
  int kthSmallest(vector<vector<int>>& matrix, int k) {
    int l = matrix[0][0];
    int r = matrix.back().back();
    while (l < r) {
      int m = l + (r - l) / 2;
      int cnt = 0;  // 记录不大于 m 的元素数
      for (auto& x : matrix) {
        cnt += upper_bound(x.begin(), x.end(), m) - x.begin();
      }
      if (k <= cnt) {
        r = m;
      } else {
        l = m + 1;
      }
    }
    return l;
  }
};
```
