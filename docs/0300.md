* 动态规划，`dp[i]` 表示以 `nums[i]` 结尾的最长子序列长度

```cpp
class Solution {
 public:
  int lengthOfLIS(vector<int>& nums) {
    if (nums.empty()) {
      return 0;
    }
    vector<int> dp(nums.size(), 1);
    int res = 1;
    for (int i = 0; i < dp.size(); ++i) {
      for (int j = 0; j < i; ++j) {
        if (nums[i] > nums[j]) {
          dp[i] = max(dp[i], dp[j] + 1);
        }
      }
      res = max(res, dp[i]);
    }
    return res;
  }
};
```

* 实现如下

```cpp
class Solution {
 public:
  int lengthOfLIS(vector<int>& nums) {
    if (nums.empty()) {
      return 0;
    }
    vector<int> v{nums[0]};
    for (int i = 1; i < nums.size(); ++i) {
      if (nums[i] > v.back()) {
        v.emplace_back(nums[i]);
      } else {
        *lower_bound(v.begin(), v.end(), nums[i]) = nums[i];
      }
    }
    return v.size();
  }
};
```
