* 求出 `0` 到 `nums.size()` 和，再减去数组中所有元素即为结果

```cpp
class Solution {
 public:
  int missingNumber(vector<int>& nums) {
    int n = nums.size();
    n = n * (n + 1) / 2;
    return accumulate(nums.begin(), nums.end(), n, minus<int>{});
  }
};
```

* 如果考虑溢出

```cpp
class Solution {
 public:
  int missingNumber(vector<int>& nums) {
    int res = nums.size();
    for (int i = 0; i < nums.size(); ++i) {
      res += i - nums[i];
    }
    return res;
  }
};
```
